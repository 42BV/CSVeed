9       private static int excelColumnToColumnIndex(String excelColumn) {
10          excelColumn = excelColumn.toUpperCase();
11          int sum = 0;
12          for (char currentChar : excelColumn.toCharArray()) {
13              sum *= 26;
14              sum += currentChar - 'A' + 1;
15          }
16          return sum;
17      }
